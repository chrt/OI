const int MOD = 10007;
int main()
{
	char c;
	int n = 0;
	while (c = getchar(), !isdigit(c)) ;
	n = c - '0';
	while (c = getchar(), isdigit(c)) n = (n*10 + c - '0') % MOD;
	printf("%lld\n", 1LL * n * (n+1) * (n+2) * 1668 % MOD);
	return 0;
}
