#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main()
{
	ll x, k;
	int T = 0;
	while (scanf("%lld%lld", &x, &k) == 2 && x) {
		ll i, y = 0;
		for (i = 1; i <= k; ++i) {
			x = (x+i-1)/i;
			if (x == y) break;
			else y = x;
			x *= i;
		}
		printf("Case #%d: %lld\n", ++T, x == y ? x*k : x);
	}
	return 0;
}
