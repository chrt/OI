typedef long long ll;

const int S = 1e5;

ll s, c[4], d[4], f[S+1];

int main()
{
	int tot;
	f[0] = 1;
	rep (i, 0, 4) {
		scanf("%lld", c+i);
		rep (j, c[i], S+1)
			f[j] += f[j-c[i]];
	}
	scanf("%d", &tot);
	while (tot--) {
		rep (i, 0, 4)
			scanf("%lld", d+i);
		scanf("%lld", &s);
		ll ans = 0;
		rep (i, 0, 1<<4) {
			ll t = s;
			int sgn = 1;
			rep (j, 0, 4)
				if (i & (1<<j))
					t -= (d[j]+1) * c[j], sgn *= -1;
			if (t >= 0) ans += sgn * f[t];
		}
		printf("%lld\n", ans);
	}
	return 0;
}
